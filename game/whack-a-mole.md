10 X=RND(7)+1:LC 0,1:? "SCORE : ";S;"   
20 GSB 210:IF N<=0 LRUN 2
30 IF (X%2)=0 OUT 1,0:GOTO 50
40 BEEP 10,5:OUT 1,1:N=N-1
50 IF ((X/2)%2)=0 OUT 2,0:GOTO 70
60 BEEP 12,5:OUT 2,1:N=N-1
70 IF ((X/4)%2)=0 OUT 3,0:GOTO 90
80 BEEP 15,5:OUT 3,1:N=N-1
90 CLT
100 IF (X%2+IN(1))*(X/2%2+IN(2))*(X/4%2+IN(4))=0 OR X=7 AND IN(1)+IN(2)+IN(4) GOTO 170 
110 IF (X%2)=0 OR IN(1) GOTO 130
120 OUT 1,0:X=X&6:BEEP 10,20:GSB 190
130 IF ((X/2)%2)=0 OR IN(2) GOTO 150
140 OUT 2,0:X=X&5:BEEP 12,20:GSB 190
150 IF ((X/4)%2)=0 OR IN(4) GOTO 170
160 OUT 3,0:X=X&3:BEEP 15,20:GSB 190
170 IF TICK()<D GOTO 100
180 OUT 0:WAIT D:GOTO 10
